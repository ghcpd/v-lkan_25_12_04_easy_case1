if (-not (Test-Path $inputPath)) { throw "Input file not found: $inputPath" }
$root = Split-Path $inputPath;
Get-Content -LiteralPath $inputPath | ForEach-Object {
    if (-not $_) { return }
    $obj = $_ | ConvertFrom-Json
    $repo = $obj.repo
    if (-not $repo) { return }
    $san = $repo -replace '/', '__'
    $dir = Join-Path $root $san
    if (-not (Test-Path $dir)) { New-Item -ItemType Directory -Path $dir | Out-Null }
    $outFile = Join-Path $dir 'repo_metadata.jsonl'
    Add-Content -LiteralPath $outFile -Value $_ -Encoding UTF8
}
Write-Host 'Done.'
